def show_thumbnail(project)
        if project.thumbnail?
                project.thumbnail.url
        elsif !project.sub_images.empty?
                project.sub_images.first.url(:thumbnail)
        else
                "http://placehold.it/300x200&text=No thumbnail."
        end
end
